🧱 4. JAVASCRIPT TRACKING FUNCTION

function trackEvent(type, business_id) {
    fetch('track.php', {
        method: 'POST',
        body: JSON.stringify({
            event_type: type,
            business_id: business_id,
            page: window.location.href
        })
    });
}
