function sendMessage() {
    let msg = document.getElementById("chat-input").value;

    appendMessage("user", msg);

    fetch("chatbot.php", {
        method: "POST",
        body: JSON.stringify({message: msg})
    })
    .then(res => res.json())
    .then(data => {
        appendMessage("bot", data.reply);
    });
}
